rsqlGetStmtString

Return the SQL statement string for a statement handle

Prototype

RSQL_ERRCODE EXTERNAL_FCN rsqlGetStmtString(
    HSTMT               hStmt, 
    char               *pStmtbuf,
    uint16_t            buflen, 
    uint16_t           *pStrlen)

Arguments

pHstmt (input) Statement handle.
pStmtbuf (output) Pointer to statement string buffer.
buflen (input) Length of statement string buffer (including null byte).
pStrlen (output) Length of statement string (not including null byte).

Description

This function can be used to get a copy of the SQL statement associated with statement handle hstmt. The statement must have been compiled through a prior call to rsqlPrepare (or rsqlExecDirect). In order to first discover the length of the statement string (e.g., so you can allocate an appropriately sized buffer) you can call rsqlGetStmtString with NULL for pStmtbuf and with pStrlen equal to the address of a uint16_t variable. Status code errTRUNCATE is returned when buflen is too small to contain the entire statement string.

Example

#include "rsql.h"
 	
/* Display execution plan for specified statement handle
*/
RSQL_ERRCODE DisplayPlan(
   HSTMT       hQuery)
{
	RSQL_ERRCODE stat;
	HCONN        hdbc;
	HSTMT        hPlan;
	RSQL_VALUE  *row;
	uint16_t     nocols;
	char        *stmtstr;
	uint16_t     strlen;
	uint64_t     plancost;
	const char  *emsg;

   /* allocate handle for hQuery exec plan result set */
   stat = rsqlGetConnHandle(hQuery, &hdbc);
   if ( stat != errSUCCESS ) goto err_handler;

   stat = rsqlAllocStmt(hdbc, &hPlan);
   if ( stat != errSUCCESS ) goto err_handler;
    
	stat = rsqlShowPlan(hPlan, hQuery);
	if ( stat != errSUCCESS ) goto err_handler;

	/* get length of stmt string */
	rsqlGetStmtString(hQuery, NULL, 0, &strlen);  	

       /* allocate stmtstr buffer */
       stmtstr = psp_getMemory(++strlen, mtag);

	/* retrieve hQuery statement string */
	rsqlGetStmtString(hQuery, stmtstr, strlen, NULL);

	printf("\nEXECUTION PLAN FOR: \n%s\n\n", stmtstr);

	/* free stmtstr buffer */
	psp_freeMemory(stmtstr, mtag);

	...
}
