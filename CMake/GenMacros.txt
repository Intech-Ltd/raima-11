
macro (ddlp_cpp_c_d_e)
  foreach (NAME ${ARGV})
    add_custom_command (
      OUTPUT
        ${CMAKE_CURRENT_BINARY_DIR}/${NAME}.h
        ${CMAKE_CURRENT_BINARY_DIR}/${NAME}_dbd.c
        ${CMAKE_CURRENT_BINARY_DIR}/${NAME}_dbd.h
        ${CMAKE_CURRENT_BINARY_DIR}/${NAME}_api.cpp
        ${CMAKE_CURRENT_BINARY_DIR}/${NAME}_api.h
        ${CMAKE_CURRENT_BINARY_DIR}/embed_${NAME}_dbd.c
        ${CMAKE_CURRENT_BINARY_DIR}/embed_${NAME}_dbd.h
        ${CMAKE_CURRENT_BINARY_DIR}/${NAME}.ddl.c
        ${CMAKE_CURRENT_BINARY_DIR}/${NAME}.ddl.h
      COMMAND native-embedfile -o ${NAME}.ddl ${CMAKE_CURRENT_SOURCE_DIR}/${NAME}.ddl
      COMMAND native-ddlp -cpp -c -d -e -B -q -nd ${CMAKE_CURRENT_SOURCE_DIR}/${NAME}.ddl
      DEPENDS 
        ${CMAKE_CURRENT_SOURCE_DIR}/${NAME}.ddl
        native-embedfile
        native-ddlp
      COMMENT "ddlp_cpp_c_d_e ${NAME}.ddl"
    )
    set (SOURCE_FILES ${SOURCE_FILES}
        ${CMAKE_CURRENT_BINARY_DIR}/${NAME}.h
        ${CMAKE_CURRENT_BINARY_DIR}/${NAME}_dbd.c
        ${CMAKE_CURRENT_BINARY_DIR}/${NAME}_dbd.h
        ${CMAKE_CURRENT_BINARY_DIR}/${NAME}_api.cpp
        ${CMAKE_CURRENT_BINARY_DIR}/${NAME}_api.h
        ${CMAKE_CURRENT_BINARY_DIR}/embed_${NAME}_dbd.c
        ${CMAKE_CURRENT_BINARY_DIR}/embed_${NAME}_dbd.h
        ${CMAKE_CURRENT_BINARY_DIR}/${NAME}.ddl.c
        ${CMAKE_CURRENT_BINARY_DIR}/${NAME}.ddl.h
    )
  endforeach (NAME)
endmacro (ddlp_cpp_c_d_e)

      COMMAND native-ddlp -c -d -e -B -q -nd ${CMAKE_CURRENT_SOURCE_DIR}/${NAME}.ddl
      DEPENDS 
        ${CMAKE_CURRENT_SOURCE_DIR}/${NAME}.ddl
        native-embedfile
        native-ddlp
      COMMENT "ddlp_c_d_e ${NAME}.ddl"
    )
    set (SOURCE_FILES ${SOURCE_FILES}
        ${CMAKE_CURRENT_BINARY_DIR}/${NAME}.h
        ${CMAKE_CURRENT_BINARY_DIR}/${NAME}_dbd.c
        ${CMAKE_CURRENT_BINARY_DIR}/${NAME}_dbd.h
        ${CMAKE_CURRENT_BINARY_DIR}/embed_${NAME}_dbd.c
        ${CMAKE_CURRENT_BINARY_DIR}/embed_${NAME}_dbd.h
        ${CMAKE_CURRENT_BINARY_DIR}/${NAME}.ddl.c
        ${CMAKE_CURRENT_BINARY_DIR}/${NAME}.ddl.h
    )
  endforeach (NAME)
endmacro (ddlp_c_d_e)
